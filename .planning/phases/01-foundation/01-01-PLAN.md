    Update `start/kernel.ts` to use the v7 three-stack middleware pattern (per Research Pattern 2):
    ```typescript
    // start/kernel.ts
    import router from '@adonisjs/core/services/router'
    import server from '@adonisjs/core/services/server'

    /**
     * Server middleware runs for ALL requests including 404s.
     * CORS and security headers must run here — before routing.
     */
    server.use([
      () => import('@adonisjs/cors/cors_middleware'),
      // Security headers middleware added in Plan 01-05
    ])

    /**
     * Router middleware runs only when a route matches.
     */
    router.use([
      () => import('@adonisjs/core/bodyparser_middleware'),
    ])

    /**
     * Named middleware — applied explicitly per route or group.
     * auth MUST be applied before tenant (TenantMiddleware reads from auth.user).
     */
    export const middleware = router.named({
      auth: () => import('@adonisjs/auth/auth_middleware'),
      // tenant and throttle added in Plan 01-05
    })
    ```
